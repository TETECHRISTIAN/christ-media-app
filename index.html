function voirHistoDette(fid){ const f = S.fournisseurs.find(x => x.id == fid); if(!f) return; // Récupérer les historiques const ajoutsRaw = f.historiqueDetteAjouts || []; const paiementsRaw = f.historiquePaiementsFour || []; // Nettoyer les doublons par ID const seenAjouts = new Set(); const ajouts = ajoutsRaw.filter(h => { const key = String(h.id); if(seenAjouts.has(key)) return false; seenAjouts.add(key); return true; }); const seenPaiements = new Set(); const paiements = paiementsRaw.filter(p => { const key = String(p.id); if(seenPaiements.has(key)) return false; seenPaiements.add(key); return true; }); // Construire les lignes pour le tableau const lignes = []; ajouts.forEach(h => { lignes.push({ date: h.date, type: 'ajout', mt: h.mt, note: h.notes, ts: h.ts || 0, id: h.id }); }); paiements.forEach(p => { lignes.push({ date: p.date, type: 'paiement', mt: p.montantPaye || p.mt || 0, note: p.note, compte: p.compte, ts: p.ts || 0, id: p.id }); }); // Trier par date puis par timestamp lignes.sort((a,b) => { const dateCompare = (a.date || '').localeCompare(b.date || ''); if(dateCompare !== 0) return dateCompare; return (a.ts || 0) - (b.ts || 0); }); // Recalculer les soldes AVANT/APRES de façon chronologique let soldeCourant = 0; lignes.forEach(l => { l.avant = soldeCourant; if(l.type === 'ajout') { soldeCourant += l.mt; } else { // Un paiement ne peut pas rendre le solde négatif soldeCourant = Math.max(0, soldeCourant - l.mt); l.effectif = Math.min(l.mt, l.avant); } l.apres = soldeCourant; }); // Calculer les totaux pour l'en-tête const totalAjoute = ajouts.reduce((s, h) => s + (h.mt || 0), 0); const totalPaye = paiements.reduce((s, p) => s + (p.montantPaye || p.mt || 0), 0); const soldeActuel = soldeCourant; // Mettre à jour la dette du fournisseur f.dette = soldeActuel; // Détection des paiements fantômes (excédentaires) const nbFantomes = lignes.filter(l => l.type === 'paiement' && l.effectif === 0 && l.mt > 0).length; // Compter les doublons potentiels const paiemSeenDM = new Set(); let nbDoublonsPaiements = 0; paiements.forEach(p => { const k = `${p.date}|${p.montantPaye || p.mt || 0}`; if(paiemSeenDM.has(k)) nbDoublonsPaiements++; else paiemSeenDM.add(k); }); modal(`
📜 Relevé de compte — ${f.nom}
TOTAL MIS EN DETTE
+${F(totalAjoute)}
${ajouts.length} ajout(s)
TOTAL PAYÉ
-${F(totalPaye)}
${paiements.length} paiement(s)
SOLDE ACTUEL
${F(soldeActuel)}
${soldeActuel === 0 ? '✅ Compte soldé' : 'À régler'}
${lignes.length === 0 ? '
Aucun mouvement enregistré
' : ` ${lignes.map(l => ` `).join('')}
DATE TYPE ANCIEN SOLDE MOUVEMENT NOUVEAU SOLDE NOTE / COMPTE
${fmtD(l.date)} ${l.type === 'ajout' ? '📈 Ajout' : `💳 Paiement ${l.effectif === 0 && l.mt > 0 ? '⚠️ Excédent' : ''} ${l.compte ? '
' + l.compte + '
' : ''}` } 		${F(l.avant)} ${l.type === 'ajout' ? '+' : '-'}${F(l.mt)} ${F(l.apres)} ${l.note || '—'}
` }
${lignes.length} mouvement(s) · ${ajouts.length} ajout(s) · ${paiements.length} paiement(s) ${nbFantomes > 0 ? ` · ⚠️ ${nbFantomes} paiement(s) excédentaire(s)` : ''}
${nbDoublonsPaiements > 0 ? ` ` : ''}
`); }
